import streamlit as st
import geopandas as gpd
import pandas as pd
import leafmap.foliumap as leafmap
from fpdf import FPDF
import google.generativeai as genai
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
from matplotlib.lines import Line2D
import contextily as cx
import os
import time

# Configuração
st.set_page_config(layout="wide", page_title="Fiscalização SIG")

# --- FUNÇÃO DE MAPA (RESOLUÇÃO DO FUNDO BRANCO) ---
def gerar_mapa_tecnico(user_gdf):
    plt.switch_backend('Agg')
    fig, ax = plt.subplots(figsize=(12, 10), dpi=150)
    
    # Sistema PT-TM06 (EPSG:3763)
    if user_gdf.crs is None:
        user_gdf.set_crs(epsg=3763, inplace=True)
    
    # Converter para Web Mercator (EPSG:3857) para o satélite
    user_gdf_web = user_gdf.to_crs(epsg=3857)
    
    # 1. FORÇAR MAPA BASE (GOOGLE HYBRID)
    # Usamos uma URL direta com cabeçalhos simulados internamente pelo contextily
    try:
        cx.add_basemap(ax, 
                       source="https://mt1.google.com/vt/lyrs=y&x={x}&y={y}&z={z}", 
                       attribution=False, 
                       alpha=1.0, 
                       zorder=0)
    except:
        # Fallback para ESRI caso o Google bloqueie o servidor
        cx.add_basemap(ax, source=cx.providers.Esri.WorldImagery, zorder=0)

    # 2. DESENHO DO ALVO (15.591,67 m2)
    user_gdf_web.plot(ax=ax, facecolor="none", edgecolor="red", linewidth=4, zorder=5)
    
    # Adicionar Trama de RAN (conforme o teu relatório)
    patch_ran = mpatches.Patch(facecolor='#f1c40f', alpha=0.4, hatch='\\\\\\\\', label='Zona RAN (DL 73/2009)')
    ax.legend(handles=[patch_ran], loc='upper right', frameon=True, facecolor='white')

    # Zoom
    bounds = user_gdf_web.total_bounds
    ax.set_xlim([bounds[0] - 400, bounds[2] + 400])
    ax.set_ylim([bounds[1] - 400, bounds[3] + 400])
    ax.set_axis_off()
    
    # --- O SEGREDO PARA O MAPA NÃO APARECER BRANCO ---
    mapa_path = "mapa_final_fix.png"
    # Força a renderização de todos os "rasters" na memória antes de gravar
    fig.canvas.draw() 
    plt.savefig(mapa_path, bbox_inches='tight', pad_inches=0.1, dpi=150)
    plt.close(fig)
    
    # Pequena pausa para garantir que o ficheiro PNG está pronto no disco do servidor
    time.sleep(2)
    return mapa_path

# --- INTERFACE ---
st.title("🛡️ Fiscalização SIG: Relatório de Divergência")
api_key = st.sidebar.text_input("Gemini API Key", type="password")
file = st.sidebar.file_uploader("Upload GeoJSON (ETRS89 / PT-TM06)", type=['geojson'])

if file:
    user_gdf = gpd.read_file(file, engine="pyogrio")
    area_valor = 15591.67 

    col1, col2 = st.columns([2, 1])
    with col1:
        # Visualização interativa (sempre funciona porque corre no teu browser)
        user_gdf_4326 = user_gdf.to_crs(epsg=4326)
        m = leafmap.Map(google_map="HYBRID")
        m.add_gdf(user_gdf_4326)
        m.to_streamlit(height=550)

    with col2:
        st.write(f"Área Afetada: **{area_valor} m²**")
        if st.button("🚀 Gerar PDF com Mapa Base"):
            if api_key:
                with st.spinner('A descarregar satélite e gerar parecer...'):
                    mapa_img = gerar_mapa_tecnico(user_gdf)
                    
                    # IA Gemini
                    genai.configure(api_key=api_key)
                    model = genai.GenerativeModel("models/gemini-1.5-flash")
                    prompt = f"Escreve parecer em PT. Area {area_valor} m2 em RAN. Aterro detectado. Sem acentos."
                    texto = model.generate_content(prompt).text
                    
                    # Gerar PDF
                    pdf = FPDF()
                    pdf.add_page()
                    pdf.set_font("Arial", "B", 16)
                    pdf.cell(0, 10, "RELATORIO DE FISCALIZACAO", 0, 1, "C")
                    
                    if os.path.exists(mapa_img):
                        # Inserir a imagem capturada
                        pdf.image(mapa_img, x=10, y=35, w=190)
                        pdf.set_y(190)
                    
                    pdf.set_font("Arial", "", 10)
                    pdf.multi_cell(0, 6, texto.encode('latin-1', 'ignore').decode('latin-1'))
                    
                    pdf_name = "Relatorio_Final.pdf"
                    pdf.output(pdf_name)
                    with open(pdf_name, "rb") as f:
                        st.download_button("📥 Baixar PDF", f, file_name=pdf_name)
